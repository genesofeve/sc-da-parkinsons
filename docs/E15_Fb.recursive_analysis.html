Calculating summary stats

# Calculating summary stats
fData(e15.Fb.dat.filter)$mean_expr<-apply(round(exprs(e15.Fb.dat.filter)),1,mean) # mean expression
fData(e15.Fb.dat.filter)$sd_expr<-apply(round(exprs(e15.Fb.dat.filter)),1,sd) # sd expression
fData(e15.Fb.dat.filter)$bcv<-(fData(e15.Fb.dat.filter)$sd_expr/fData(e15.Fb.dat.filter)$mean_expr)**2 # calculating biological coefficient of variation
fData(e15.Fb.dat.filter)$percent_detection<-(fData(e15.Fb.dat.filter)$num_cells_expressed/dim(e15.Fb.dat.filter)[2])*100 # calculating % detection

Identifying clusters

Identifying clusters in the data in an unsupervised manner

# Going to attempt to use the R program 'ADPclust' to determine how many
# clusters our data has

# Loading NbClust
library(ADPclust)

# Running ADPclust
clust.res <- adpclust(x = e15.Fb.dat.filter.BCV.tsne, draw = T)

# Extracting the 'best partition' (aka the best cluster) for each cell
clust.res.df <- as.data.frame(clust.res$cluster)

# Adding the cluster assignment for each cell to the pData
pData(e15.Fb.dat.filter)$kmeans_tSNE_cluster <- as.factor(clust.res.df$`clust.res$cluster`)

# Plotting the same tSNE plot as above but coloring with the 'clusters'
myTSNEPlotAlpha(e15.Fb.dat.filter, color = "kmeans_tSNE_cluster", shape = "age") + 
    scale_color_brewer(palette = "Set1") + ggtitle("All Cells - BCV tSNE with Clusters Plot")
