# Plot number of cells expressing each gene as histogram
hist(fData(e15.Fb.dat.filter)$num_cells_expressed,breaks=50,col="red",main="Cells expressed per gene")

# Keep only expressed genes with expression in >= 5% of cells
numCellThreshold<-nrow(pData(e15.Fb.dat.filter))*0.05
e15.Fb.dat.expressed_genes<-row.names(subset(fData(e15.Fb.dat.filter),num_cells_expressed >= numCellThreshold))

# Same plot as above with threshold
hist(fData(e15.Fb.dat.filter)$num_cells_expressed,breaks=50,col="red",main="Cells expressed per gene - threshold")

# Only keeping "expressed" genes
e15.Fb.dat.filter <-e15.Fb.dat.filter[e15.Fb.dat.expressed_genes,]

# Estimating the size factors
e15.Fb.dat.filter <-estimateSizeFactors(e15.Fb.dat.filter)

# Estimating dispersions
e15.Fb.dat.filter <- estimateDispersions(e15.Fb.dat.filter,cores=8)

# Calculating summary stats
fData(e15.Fb.dat.filter)$mean_expr<-apply(round(exprs(e15.Fb.dat.filter)),1,mean) # mean expression
fData(e15.Fb.dat.filter)$sd_expr<-apply(round(exprs(e15.Fb.dat.filter)),1,sd) # sd expression
fData(e15.Fb.dat.filter)$bcv<-(fData(e15.Fb.dat.filter)$sd_expr/fData(e15.Fb.dat.filter)$mean_expr)**2 # calculating biological coefficient of variation
fData(e15.Fb.dat.filter)$percent_detection<-(fData(e15.Fb.dat.filter)$num_cells_expressed/dim(e15.Fb.dat.filter)[2])*100 # calculating % detection
e15.Fb.dat.filter.genes <- e15.Fb.dat.filter # spoofing the CellDataSet
disp_table <- dispersionTable(e15.Fb.dat.filter.genes) # pulling out the dispersion table
unsup_clustering_genes <-subset(disp_table, mean_expression >= 0.5 & dispersion_empirical >= 2 * dispersion_fit) # subsetting the data to pull out genes with expression above 0.5 and dispersion empirical > 2
e15.Fb.dat.high_bcv_genes<-unsup_clustering_genes$gene_id # pulling out list of genes
e15.Fb.dat.filter.order <- setOrderingFilter(e15.Fb.dat.filter, unsup_clustering_genes$gene_id)
plot_ordering_genes(e15.Fb.dat.filter.order) # plotting the dispersion and genes
